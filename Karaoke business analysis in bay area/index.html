Featured SQL Query — Revenue by City
SELECT city, COUNT(*) AS total_bookings, ROUND(SUM(total_revenue), 2) AS total_revenue, ROUND(AVG(per_person_spend), 2) AS avg_per_person, ROUND(SUM(total_revenue) / COUNT(DISTINCT venue_id), 2) AS revenue_per_venue FROM bookings GROUP BY city ORDER BY total_revenue DESC;

Weekend vs Weekday Query
SELECT CASE WHEN is_weekend THEN 'Weekend' ELSE 'Weekday' END AS day_type, COUNT(*) AS bookings, ROUND(AVG(total_revenue), 2) AS avg_revenue, ROUND(SUM(total_revenue) * 100.0 / (SELECT SUM(total_revenue) FROM bookings), 1) AS revenue_share_pct FROM bookings GROUP BY is_weekend;
YoY Growth — Window Function
WITH yearly AS ( SELECT city, year, SUM(total_revenue) AS revenue FROM bookings GROUP BY city, year ) SELECT a.city, a.year, ROUND(a.revenue, 2) AS revenue, ROUND((a.revenue - b.revenue) * 100.0 / b.revenue, 1) AS yoy_growth_pct FROM yearly a LEFT JOIN yearly b ON a.city = b.city AND a.year = b.year + 1;
